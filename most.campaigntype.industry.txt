WITH ranked AS (
    SELECT
        campaign_type,
        industry,
        COUNT(*) AS industry_count,
        ROW_NUMBER() OVER (
            PARTITION BY campaign_type
            ORDER BY COUNT(*) DESC
        ) AS rn
    FROM ads_performance
    GROUP BY campaign_type, industry
)
SELECT 
    campaign_type,
    industry AS most_campaign_type_industry,
    industry_count
FROM ranked
WHERE rn = 1;
