SELECT
    country,
    SUM(impressions) AS impressions,
    SUM(clicks) AS clicks,
    SUM(ad_spend) AS spend,
    SUM(conversions) AS conversions,
    SUM(revenue) AS revenue,

    AVG(CTR) AS CTR,
    AVG(CPC)AS  CPC,
    AVG(CPA) AS CPA,
    AVG(ROAS)AS ROAS

FROM Ads_performance
GROUP BY country
ORDER BY revenue DESC;
